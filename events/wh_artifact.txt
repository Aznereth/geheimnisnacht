
# Become Lunatic
character_event = {
	id = crown_blade.1
	desc = EVTDESCcrown_blade.1_ARTIFACT
	picture = GFX_evt_lunatic
	border = GFX_event_normal_frame_religion
	is_triggered_only = yes
	only_playable = yes

	min_age = 4

	trigger = {
		NOT = {
			trait = lunatic
		}
		NOT = {
			is_undead_trigger = yes
		}
		NOT = {
			is_demon_trigger = yes
		}
		OR = {
			# Senile/comatose due to old age do not gain lunatic, immortals get a pass though
			NOT = {
				age = 65
			}
			immortal = yes
		}
	}

	immediate = {
		FROM = {
			OR = {
				has_artifact = nagash_crown
				has_artifact = alcadizar's_blade
			}
		}
	}

	option = {
		name = {
			text = EVTOPTAcrown_blade.1_ARTIFACT
			trigger = {
				OR = {
					has_artifact = nagash_crown
					has_artifact = alcadizar's_blade
				}
			}
		}
			add_trait = lunatic
	}
}
